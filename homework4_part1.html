## Gather the pups data to long format

pups_data <- read_csv("./FAS_pups.csv") %>% ## Load the data
              clean_names() %>% ## Clean Names
              mutate(id = 1:313) %>% ## Create Pup ID
              gather(key = outcome, 
                     value = post_natal_days, pd_ears:pd_walk) %>% ## Gather to long format
              separate(outcome, into = c("remove", "outcome"), sep = "_") %>% ## Clean names by separate
              select(-remove) ## Remove dummy variable
